 16class Gordon(BaseMultilabelClassifier):
 17    """
 18    Fusion model inspired by
 19
 20        Y. Liu, H. -P. Lu and C. -H. Lai, "A Novel Attention-Based Multi-Modal
 21        Modeling Technique on Mixed Type Data for Improving TFT-LCD Repair
 22        Process," in IEEE Access, vol. 10, pp. 33026-33036, 2022, doi:
 23        10.1109/ACCESS.2022.3158952.
 24
 25    In a nutshell, the tabular data is encoded by a MLP as usual. The image, on
 26    the other hand, goes through several
 27    `acc23.models.imagetabnet.VisionEncoder`, which also involves the encoded
 28    tabular data. The result of both branches are concatenated and fed through
 29    a MLP head.
 30    """
 31
 32    _tab_mlp: nn.Module
 33    _vis_pool: nn.Module
 34    _vis_enc: nn.Module
 35    _vis_proj: nn.Module
 36    _mlp_head: nn.Module
 37
 38    def __init__(
 39        self,
 40        n_features: int = N_FEATURES,
 41        image_shape: Tuple[int, int, int] = (
 42            N_CHANNELS,
 43            IMAGE_SIZE,
 44            IMAGE_SIZE,
 45        ),
 46        out_dim: int = N_TRUE_TARGETS,
 47        embed_dim: int = 16,
 48        dropout: float = 0.1,
 49        activation: str = "gelu",
 50        **kwargs: Any,
 51    ) -> None:
 52        """
 53        Args:
 54            n_features (int, optional): Number of numerical tabular features
 55            image_shape (Tuple[int, int, int], optional):
 56            out_dim (int, optional):
 57            embed_dim (int, optional):
 58            dropout (float, optional):
 59            activation (str, optional):
 60
 61        See also:
 62            `acc23.models.base_mlc.BaseMultilabelClassifier.__init__`
 63        """
 64        super().__init__(**kwargs)
 65        self.save_hyperparameters()
 66        nc, s, _ = image_shape
 67        self._tab_mlp = MLP(
 68            n_features,
 69            [embed_dim],
 70            activation=activation,
 71        )
 72        self._vis_pool = nn.MaxPool2d(5, 2, 2)
 73        # self.vision_branch_a = nn.Identity()
 74        self._vis_enc = VisionEncoder(
 75            # image_shape=image_shape,
 76            image_shape=(nc, s // 2, s // 2),
 77            out_channels=[
 78                # 4,  # 512 -> 256
 79                4,  # 256 -> 128
 80                4,  # -> 64
 81                4,  # -> 32
 82                4,  # -> 16
 83                4,  # -> 8
 84                4,  # -> 4
 85            ],
 86            in_features=embed_dim,
 87            activation=activation,
 88        )
 89        self._vis_proj = nn.Linear(4 * 4 * 4, embed_dim, bias=False)
 90        # self.main_branch = nn.Sequential(
 91        #     nn.Linear(2 * embed_dim, out_dim),
 92        # )
 93        self._mlp_head = nn.Sequential(
 94            nn.LayerNorm(embed_dim),
 95            nn.Linear(embed_dim, embed_dim),
 96            get_activation(activation),
 97            nn.Dropout1d(dropout),
 98            nn.Linear(embed_dim, out_dim),
 99        )
100        self.example_input_array = (
101            torch.zeros((32, n_features)),
102            torch.zeros((32, nc, s, s)),
103        )
104        self.forward(*self.example_input_array)
105
106    def forward(
107        self,
108        x: Union[Tensor, Dict[str, Tensor]],
109        img: Tensor,
110        *_,
111        **__,
112    ) -> Tensor:
113        """
114        Args:
115            x (Tensor): Tabular data with shape `(N, N_FEATURES)`, where `N` is
116                the batch size, or alternatively, a string dict, where each key
117                is a `(N,)` tensor.
118            img (Tensor): Batch of images, i.e. a tensor of shape
119                `(N, N_CHANNELS, IMAGE_SIZE, IMAGE_SIZE)`
120        """
121        if isinstance(x, dict):
122            x = concat_tensor_dict(x)
123        x = x.float().to(self.device)  # type: ignore
124        img = img.to(self.device)  # type: ignore
125        u = self._tab_mlp(x)
126        v = self._vis_pool(img)
127        v = self._vis_enc(v, u)
128        v = self._vis_proj(v)
129        # uv = torch.concatenate([u, v], dim=-1)
130        uv = u * v
131        w = self._mlp_head(uv)
132        return w

example_input_array: Union[torch.Tensor, Tuple, Dict, NoneType]

The example input array is a specification of what the module can consume in the forward() method. The return type is interpreted as follows:

  • Single tensor: It is assumed the model takes a single argument, i.e., model.forward(model.example_input_array)
  • Tuple: The input array should be interpreted as a sequence of positional arguments, i.e., model.forward(*model.example_input_array)
  • Dict: The input array represents named keyword arguments, i.e., model.forward(**model.example_input_array)
