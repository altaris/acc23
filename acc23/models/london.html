 30        self,
 31        n_features: int = N_FEATURES,
 32        image_shape: Tuple[int, int, int] = (
 33            N_CHANNELS,
 34            IMAGE_SIZE,
 35            IMAGE_SIZE,
 36        ),
 37        out_dim: int = N_TRUE_TARGETS,
 38        embed_dim: int = 128,
 39        patch_size: int = 8,
 40        n_transformers: int = 16,
 41        n_heads: int = 8,
 42        dropout: float = 0.1,
 43        activation: str = "gelu",
 44        mlp_dim: int = 256,
 45        vit: Literal["new", "pretrained", "frozen"] = "pretrained",
 46        pooling: bool = True,
 47        **kwargs: Any,
 48    ) -> None:
 49        """
 50        Args:
 51            n_features (int, optional): Number of numerical tabular features
 52            image_shape (Tuple[int, int, int], optional):
 53            out_dim (int, optional):
 54            embed_dim (int, optional):
 55            patch_size (int, optional):
 56            n_transformers (int, optional): Ignored if `vit` is not `new`
 57            n_heads (int, optional): Ignored if `vit` is not `new`
 58            dropout (float, optional):
 59            activation (str, optional):
 60            mlp_dim (int, optional):
 61            vit (Literal["new", "pretrained", "frozen"], optional): How to
 62                create the vision transformer:
 63                - `new`: a new vision transformer is created (see [Hugging Face
 64                Transformer's
 65                `ViTModel`s](https://huggingface.co/docs/transformers/v4.30.0/en/model_doc/vit#transformers.ViTModel))
 66                - `pretrained`: a pretrained `ViTModel` is used (specifically
 67                `google/vit-base-patch16-224`)
 68                - `frozen`: same, but the ViT is frozen
 69
 70        See also:
 71            `acc23.models.base_mlc.BaseMultilabelClassifier.__init__`
 72        """
 73        super().__init__(**kwargs)
 74        self.save_hyperparameters()
 75        nc, s, _ = image_shape
 76        self._tab_mlp = MLP(
 77            in_dim=n_features,
 78            hidden_dims=[mlp_dim, embed_dim],
 79            dropout=dropout,
 80            activation=activation,
 81            is_head=False,
 82        )
 83        self._pool = nn.MaxPool2d(5, 1, 2) if pooling else nn.Identity()
 84        if vit == "new":
 85            self._vit = ViTModel(
 86                config=ViTConfig(
 87                    hidden_size=embed_dim,
 88                    num_hidden_layers=n_transformers,
 89                    num_attention_heads=n_heads,
 90                    intermediate_size=mlp_dim,
 91                    hidden_act=activation,
 92                    hidden_dropout_prob=dropout,
 93                    attention_probs_dropout_prob=dropout,
 94                    image_size=s,
 95                    num_channels=nc,
 96                    patch_size=patch_size,
 97                ),
 98                add_pooling_layer=False,
 99            )
100            self._vit_resize = nn.Identity()
101            self._vit_proj = nn.Identity()
102        else:
103            self._vit = ViTModel.from_pretrained("google/vit-base-patch16-224")
104            self._vit.requires_grad_(vit != "frozen")
105            if self._vit.pooler is not None:
106                self._vit.pooler.requires_grad_(False)
107            if not isinstance(self._vit, ViTModel):
108                raise RuntimeError(
109                    "Pretrained ViT is not a transformers.ViTModel object"
110                )
111            if not isinstance(self._vit.config, ViTConfig):
112                raise RuntimeError(
113                    "Pretrained ViT confit is not a transformers.ViTConfig "
114                    "object"
115                )
116            self._vit_resize = Resize(
117                self._vit.config.image_size, antialias=True
118            )
119            self._vit_proj = nn.Linear(
120                self._vit.config.hidden_size, embed_dim, bias=False
121            )
122        self._mlp_head = MLP(
123            in_dim=2 * embed_dim,
124            hidden_dims=[mlp_dim, out_dim],
125            dropout=dropout,
126            activation=activation,
127            is_head=True,
128        )
129        self.example_input_array = (
130            torch.zeros((32, n_features)),
131            torch.zeros((32, nc, s, s)),
132        )
133        self.forward(*self.example_input_array)

135    def forward(
136        self,
137        x: Union[Tensor, Dict[str, Tensor]],
138        img: Tensor,
139        *_,
140        **__,
141    ) -> Tensor:
142        """
143        Args:
144            x (Tensor): Tabular data with shape `(N, N_FEATURES)`, where `N` is
145                the batch size, or alternatively, a string dict, where each key
146                is a `(N,)` tensor.
147            img (Tensor): Batch of images, i.e. a tensor of shape
148                `(N, N_CHANNELS, IMAGE_SIZE, IMAGE_SIZE)`
149        """
150        # One operation per line for easier troubleshooting
151        if isinstance(x, dict):
152            x = concat_tensor_dict(x)
153        x = x.float().to(self.device)  # type: ignore
154        img = img.to(self.device)  # type: ignore
155        a = self._tab_mlp(x)
156        b = self._pool(img)
157        b = self._vit_resize(b)
158        b = self._vit(b).last_hidden_state[:, 0]
159        b = self._vit_proj(b)
160        ab = torch.concatenate([a, b], dim=-1)
161        c = self._mlp_head(ab)
162        return c

example_input_array: Union[torch.Tensor, Tuple, Dict, NoneType]

The example input array is a specification of what the module can consume in the forward() method. The return type is interpreted as follows:

  • Single tensor: It is assumed the model takes a single argument, i.e., model.forward(model.example_input_array)
  • Tuple: The input array should be interpreted as a sequence of positional arguments, i.e., model.forward(*model.example_input_array)
  • Dict: The input array represents named keyword arguments, i.e., model.forward(**model.example_input_array)
