100    def evaluate(
101        self,
102        x: Dict[str, Tensor],
103        y: Dict[str, Tensor],
104        img: Tensor,
105        stage: Optional[str] = None,
106    ) -> Tensor:
107        """
108        Computes various scores and losses:
109        1. The loss used for training,
110        2. Accuracy score,
111        3. Hamming loss,
112        4. Precision score,
113        5. Recall score,
114        6. F1 score,
115
116        Furthermore, if `stage` is given, these values are also logged to
117        tensorboard under `<stage>/loss`, `<stage>/acc`, `<stage>/ham` (lol),
118        `<stage>/prec`, `<stage>/rec`, `<stage>/f1`, and `<stage>/extra`
119        respectively.
120        """
121        y_pred = self(x, img)
122        y_true = concat_tensor_dict(y)
123        y_true = torch.where(  # Replace NaN targets
124            y_true.isnan(),
125            y_pred > 0.0,  # Predicted targets ...
126            # torch.randint_like(y_true, 2), # ... OR Random targets
127            y_true,
128        )
129        y_true = y_true.float().to(y_pred.device)  # type: ignore
130
131        # Loss
132        lfn = self.hparams["loss_function"]
133        if lfn == "bce":
134            loss = nn.functional.binary_cross_entropy_with_logits(
135                y_pred, y_true
136            )
137        elif lfn == "irlbl_bce":
138            loss = nn.functional.binary_cross_entropy_with_logits(
139                y_pred,
140                y_true,
141                weight=Tensor(TRUE_TARGETS_IRLBL).to(y_pred.device),
142            )
143        elif lfn == "focal":
144            loss = focal_loss_with_logits(y_pred, y_true)
145        elif lfn == "db":
146            p = torch.tensor(TRUE_TARGETS_PREVALENCE, device=y_pred.device)
147            loss = distribution_balanced_loss_with_logits(y_pred, y_true, p)
148        elif lfn == "mse":
149            loss = nn.functional.mse_loss(y_pred.sigmoid(), y_true)
150        elif lfn == "mc":
151            loss = mc_loss(y_pred.sigmoid(), y_true)
152        else:
153            raise ValueError(f"Unsupported loss function '{lfn}'")
154        if stage is not None:
155            kw = {
156                "y_true": y_true.cpu().detach().numpy().astype(int),
157                "y_pred": (y_pred.cpu().detach().numpy() > 0).astype(int),
158                "average": "macro",
159                "zero_division": 0,
160            }
161            self.log_dict(
162                {
163                    f"{stage}/loss": loss,
164                    f"{stage}/f1": f1_score(**kw),
165                },
166                sync_dist=True,
167                prog_bar=True,
168            )
169            self.log_dict(
170                {
171                    f"{stage}/acc": accuracy_score(kw["y_true"], kw["y_pred"]),
172                    f"{stage}/ham": hamming_loss(kw["y_true"], kw["y_pred"]),
173                    f"{stage}/prec": precision_score(**kw),
174                    f"{stage}/rec": recall_score(**kw),
175                },
176                sync_dist=True,
177            )
178        return loss

232        key: str = "train/weights",

236            every_n_epochs (int, optional):
237            key (str, optional): Key under which the histogram will be logged

242        self._key = key

244    def on_train_epoch_end(
245        self, trainer: pl.Trainer, pl_module: pl.LightningModule
246    ) -> None:
247        """Override"""
248        if trainer.current_epoch % self._every_n_epochs != 0:
249            return
250        ps = [p.flatten() for p in pl_module.parameters()]
251        if isinstance(trainer.logger, pl.loggers.TensorBoardLogger):
252            trainer.logger.experiment.add_histogram(
253                self._key,
254                torch.concat(ps),
255                bins="auto",
256                global_step=trainer.global_step,
257            )
258        elif not self._warned_wrong_logger:
259            logging.warning(
260                "ModuleWeightsHistogram callback: Trainer's logger is has "
261                f"type '{type(trainer.logger)}', but a tensorboard logger is "
262                "required. This warning will only be logged once"
263            )
264            self._warned_wrong_logger = True

267def bp_mll_loss(y_pred: Tensor, y_true: Tensor, *_, **__) -> Tensor:
268    """
269    BM-MLL loss introduced in
270
271        Zhang, Min-Ling, and Zhi-Hua Zhou. "Multilabel neural networks with
272        applications to functional genomics and text categorization." IEEE
273        transactions on Knowledge and Data Engineering 18.10 (2006): 1338-1351.
274
275    `y_pred` is expected to contain class probabilities, **not** logits.
276    """
277    y_bar = 1 - y_true
278    k = 1 / (y_true.sum(dim=-1) * y_bar.sum(dim=-1) + 1e-10)
279    s = y_true.unsqueeze(-1) * y_bar.unsqueeze(1)
280    # s_ijk = y_true_ij and (not y_true_ik)
281    a = y_pred.unsqueeze(-1) - y_pred.unsqueeze(1)
282    a = a.sigmoid()
283    # a_ijk = y_pred_ij - y_pred_ik
284    b = s * torch.exp(-a)
285    return (k * b.sum((1, 2))).mean()
286    # Some tests:
287    # y_true = torch.Tensor([[0, 1, 0], [1, 1, 0]])
288    # y_pred = torch.Tensor([[.1, .8, .9], [.1, .9, .1]])
289    # for i in range(2):
290    #     for j in range(3):
291    #         for k in range(3):
292    #             assert s[i, j, k] == y_true[i, j] * (1 - y_true[i, k])
293    #             assert a[i, j, k] == y_pred[i, j] - y_pred[i, k]

296def class_balanced_focal_loss_with_logits(
297    y_pred: Tensor,
298    y_true: Tensor,
299    count: Tensor,
300    beta: float = 0.9,
301    gamma: float = 2.0,
302) -> Tensor:
303    """
304    Implementation of the class-balanced focal loss of
305
306        Y. Cui, M. Jia, T. -Y. Lin, Y. Song and S. Belongie, "Class-Balanced
307        Loss Based on Effective Number of Samples," 2019 IEEE/CVF Conference on
308        Computer Vision and Pattern Recognition (CVPR), Long Beach, CA, USA,
309        2019, pp. 9260-9269, doi: 10.1109/CVPR.2019.00949.
310    """
311    r = (1 - beta) / (1 - beta**count + 1e-5)
312    p = y_pred.sigmoid()
313    foc = (y_true * (1 - p) + (1 - y_true) * p) ** gamma
314    bce = nn.functional.binary_cross_entropy_with_logits(
315        y_pred, y_true, reduction="none", weight=r
316    )
317    return torch.mean(foc * bce)

320def distribution_balanced_loss_with_logits(
321    y_pred: Tensor,
322    y_true: Tensor,
323    prevalence: Tensor,
324    alpha: float = 0.1,
325    beta: float = 10.0,
326    mu: float = 0.3,
327    gamma: float = 1.0,
328    lambda_: float = 5.0,
329    kappa: float = 5e-2,
330) -> Tensor:
331    """
332    Distribution-balanced loss $\\mathcal{L}_{DB}$ from
333
334        Wu, T., Huang, Q., Liu, Z., Wang, Y., Lin, D. (2020).
335        Distribution-Balanced Loss for Multi-label Classification in
336        Long-Tailed Datasets. In: Vedaldi, A., Bischof, H., Brox, T., Frahm,
337        JM. (eds) Computer Vision - ECCV 2020. ECCV 2020. Lecture Notes in
338        Computer Science(), vol 12349. Springer, Cham.
339        https://doi.org/10.1007/978-3-030-58548-8_10
340
341    It is essentially a mix between focal loss
342    (`acc23.models.base_mlc.focal_loss_with_logits`) and rebalanced cross
343    entropy
344    (`acc23.models.base_mlc.rebalanced_binary_cross_entropy_with_logits`).
345
346    See also:
347
348        Huang, Yi, et al. "Balancing methods for multi-label text
349        classification with long-tailed class distribution." arXiv preprint
350        arXiv:2109.04712 (2021).
351
352    """
353    nu = kappa * torch.log(1 / (prevalence + 1e-5) - 1 + 1e-5)
354    q_ns = torch.where(y_true == 1, 1, lambda_) * (y_pred - nu)
355    q = q_ns.sigmoid()
356    foc = torch.where(y_true == 1, 1 - q, q) ** gamma
357    lmb = torch.where(y_true == 1, 1, 1 / (lambda_ + 1e-5))
358    r_pi = torch.sum(y_true / (prevalence + 1e-5), dim=-1)
359    r = 1 / (torch.outer(r_pi, prevalence) + 1e-5)
360    r_hat = alpha + torch.sigmoid(beta * (r - mu))
361    bce = nn.functional.binary_cross_entropy_with_logits(
362        q_ns, y_true, reduction="none"
363    )
364    return (r_hat * lmb * foc * bce).sum(-1).mean()

367def focal_loss_with_logits(
368    y_pred: Tensor,
369    y_true: Tensor,
370    gamma: float = 2.0,
371) -> Tensor:
372    """
373    Balanced-focal loss of
374
375        T. -Y. Lin, P. Goyal, R. Girshick, K. He and P. Dollr, "Focal Loss for
376        Dense Object Detection," 2017 IEEE International Conference on Computer
377        Vision (ICCV), Venice, Italy, 2017, pp. 2999-3007, doi:
378        10.1109/ICCV.2017.324.
379
380    See also:
381
382        Huang, Yi, et al. "Balancing methods for multi-label text
383        classification with long-tailed class distribution." arXiv preprint
384        arXiv:2109.04712 (2021).
385
386    """
387    p = y_pred.sigmoid()
388    foc = torch.where(y_true == 1, 1 - p, p) ** gamma
389    bce = nn.functional.binary_cross_entropy_with_logits(
390        y_pred, y_true, reduction="none"
391    )
392    return (foc * bce).mean()

395def rebalanced_bce_with_logits(
396    y_pred: Tensor,
397    y_true: Tensor,
398    prevalence: Tensor,
399    alpha: float = 0.1,
400    beta: float = 10.0,
401    mu: float = 0.3,
402) -> Tensor:
403    """
404    Rebalanced binary cross-entropy loss $\\mathcal{L}_{R-BCE}$ from
405
406        Wu, T., Huang, Q., Liu, Z., Wang, Y., Lin, D. (2020).
407        Distribution-Balanced Loss for Multi-label Classification in
408        Long-Tailed Datasets. In: Vedaldi, A., Bischof, H., Brox, T., Frahm,
409        JM. (eds) Computer Vision - ECCV 2020. ECCV 2020. Lecture Notes in
410        Computer Science(), vol 12349. Springer, Cham.
411        https://doi.org/10.1007/978-3-030-58548-8_10
412
413    """
414    # prevalence: (C,), pc: (C,), pi: (N,) => r: (N, C)
415    # n_targets = y_true.shape[1]
416    # pc = 1 / (n_targets * prevalence + 1e-5)
417    # pi = torch.sum(y_true / (n_targets * prevalence + 1e-5), dim=-1)
418    # r = torch.outer(1 / (pi + 1e-5), pc)
419    r_pi = torch.sum(y_true / (prevalence + 1e-5), dim=-1)
420    r = 1 / (torch.outer(r_pi, prevalence) + 1e-5)
421    r_hat = alpha + torch.sigmoid(beta * (r - mu))
422    bce = nn.functional.binary_cross_entropy_with_logits(
423        y_pred, y_true, reduction="none"
424    )
425    return (r_hat * bce).mean()

428def to_hierarchical_logits(
429    y: Tensor, mode: Literal["min", "max", "prod"] = "max"
430) -> Tensor:
431    """
432    Given a `(N, N_TRUE_TARGETS)` tensor of true target logits, corrects some
433    of them based on the inherent hierarchy of the targets. The parent/child
434    (aka. superclass/subclass) relationship is as follows:
435    0. `Allergy_Present` parent of targets 1, 2, 3, 4
436    1. `Severe_Allergy`
437    2. `Respiratory_Allergy` parent of targets 5, 6, 7, 8, 9, 10, 11, 12,
438        13
439    3. `Food_Allergy` parent of targets 14, 15, 16, 17, 18, 19, 20, 21, 22,
440        23, 24
441    4. `Venom_Allergy` parent of targets 25, 26
442    5. `Type_of_Respiratory_Allergy_ARIA`
443    6. `Type_of_Respiratory_Allergy_CONJ`
444    7. `Type_of_Respiratory_Allergy_GINA`
445    8. `Type_of_Respiratory_Allergy_IGE_Pollen_Gram`
446    9. `Type_of_Respiratory_Allergy_IGE_Pollen_Herb`
447    10. `Type_of_Respiratory_Allergy_IGE_Pollen_Tree`
448    11. `Type_of_Respiratory_Allergy_IGE_Dander_Animals`
449    12. `Type_of_Respiratory_Allergy_IGE_Mite_Cockroach`
450    13. `Type_of_Respiratory_Allergy_IGE_Molds_Yeast`
451    14. `Type_of_Food_Allergy_Aromatics`
452    15. `Type_of_Food_Allergy_Egg`
453    16. `Type_of_Food_Allergy_Fish`
454    17. `Type_of_Food_Allergy_Fruits_and_Vegetables`
455    18. `Type_of_Food_Allergy_Mammalian_Milk`
456    19. `Type_of_Food_Allergy_Oral_Syndrom`
457    20. `Type_of_Food_Allergy_Other_Legumes`
458    21. `Type_of_Food_Allergy_Peanut`
459    22. `Type_of_Food_Allergy_Shellfish`
460    23. `Type_of_Food_Allergy_TPO`
461    24. `Type_of_Food_Allergy_Tree_Nuts`
462    25. `Type_of_Venom_Allergy_ATCD_Venom`
463    26. `Type_of_Venom_Allergy_IGE_Venom`
464
465    The argument `mode` indicates how the logits are corrected.
466    * If `mode` is `min`, child logits are capped by that of their parent:
467        $$c' = \\mathrm{min} (c, p)$$ where where $p$ and $c$ are the logits
468        of the parent and child target, and where $c'$ is the new child
469        logit.
470    * If `mode` is `max`, parent logits are corrected to be at least the
471        max of their child: $$p' = \\mathrm{max} (p, c_1, c_2, \\ldots)$$
472        where $p'$ is the new parent logit, and $c_1, c_2, \\ldots$ are the
473        child logits.
474    * If `mode` is `prod`, child logits are weighed by their parent using
475        the following formula $$c' = \\sigma^{-1} ( \\sigma (p) \\sigma (c) )$$
476
477    """
478
479    def _mask(idxs: Iterable[int]) -> Tensor:
480        """
481        Produces a `(n_true_targets,)` binary tensor m, where `m_i` is 1 iff
482        `i` is in `idx`. The tensor is cast to `float` and moved to the the
483        same device as `y`
484        """
485        m = list(map(lambda i: int(i in idxs), range(n_true_targets)))
486        return torch.tensor(m, dtype=torch.float32, device=y.device)
487
488    # (idx of parent, idx of first child, idx of last child) For example (0, 1,
489    # 4) means that target 0 (Allergy_Present) is the parent of targets 1 to 4
490    # (Severe_Allergy, Respiratory_Allergy, Food_Allergy, and Venom_Allergy).
491    # Children are always contiguous
492    hierarchy = [  # topmost parent to bottommost children
493        (0, 1, 4),
494        (2, 5, 13),
495        (3, 14, 24),
496        (4, 25, 26),
497    ]
498    if mode == "max":  # need to start with the bottommost children
499        hierarchy = hierarchy[::-1]
500    n_true_targets = y.shape[-1]
501    for p, c1, c2 in hierarchy:
502        mc = _mask(range(c1, c2 + 1))  # mc_j = 1 if j child of p
503        if mode == "min":
504            a, b = y[:, [p]].repeat(1, n_true_targets), mc * y
505            c = torch.minimum(a, b)
506            y = c + (1 - mc) * y  # c is already masked
507        elif mode == "max":
508            mp = _mask([p])  # mp_j = 1 if j = p
509            a, b = y[:, p], (mc * y).max(dim=1).values
510            c = torch.maximum(a, b).unsqueeze(-1)
511            y = mp * c + (1 - mp) * y
512        elif mode == "prod":
513            a, b = y[:, [p]].repeat(1, n_true_targets), mc * y
514            c = -torch.log(torch.exp(-a) + torch.exp(-b) + torch.exp(-a - b))
515            # c = \\sigma^{-1} (\\sigma (a) \\sigma (b))
516            y = c + (1 - mc) * y  # c is already masked
517        else:
518            raise ValueError(
519                f"Unsupported logit correction mode '{mode}'. Available modes "
520                "are 'max', 'min', and 'prod'"
521            )
522    return y

525def mc_loss(y_pred: Tensor, y_true: Tensor) -> Tensor:
526    """
527    Max constraint loss of
528
529        Giunchiglia, Eleonora, and Thomas Lukasiewicz. "Coherent hierarchical
530        multi-label classification networks." Advances in neural information
531        processing systems 33 (2020): 9662-9673.
532
533    `y_pred` is expected to contain class probabilities, **not** logits.
534    """
535
536    def _mask(idxs: Iterable[int]) -> Tensor:
537        """
538        Produces a `(n_true_targets,)` binary tensor m, where `m_i` is 1 iff
539        `i` is in `idx`. The tensor is cast to `float` and moved to the the
540        same device as `y_pred`
541        """
542        m = list(map(lambda i: int(i in idxs), range(n_true_targets)))
543        return torch.tensor(m, dtype=torch.float32, device=y_pred.device)
544
545    n_true_targets = y_pred.shape[-1]
546    hierarchy = [
547        (0, 1, 4),
548        (2, 5, 13),
549        (3, 14, 24),
550        (4, 25, 26),
551    ]
552    mcm = torch.zeros_like(y_pred, device=y_pred.device)
553    mhy = torch.zeros_like(y_pred, device=y_pred.device)
554    for p, c1, c2 in hierarchy:
555        mp = _mask([p])  # mp_j = 1 if j = p
556        mc = _mask(range(c1, c2 + 1))  # mc_j = 1 if j child of p

560        # b_i = max (y_true_ij * y_pred_ij) for j child of p
561        mcm = mcm + mp * a.unsqueeze(-1)
562        mhy = mhy + mp * b.unsqueeze(-1)
563    return -torch.where(
564        y_true == 1, torch.log(mhy + 1e-5), torch.log(1 - mcm + 1e-5)
565    ).mean()

463    25. `Type_of_Venom_Allergy_ATCD_Venom`
464    26. `Type_of_Venom_Allergy_IGE_Venom`
465

467    * If `mode` is `min`, child logits are capped by that of their parent:
468        $$c' = \\mathrm{min} (c, p)$$ where where $p$ and $c$ are the logits
469        of the parent and child target, and where $c'$ is the new child
470        logit.
471    * If `mode` is `max`, parent logits are corrected to be at least the
472        max of their child: $$p' = \\mathrm{max} (p, c_1, c_2, \\ldots)$$
473        where $p'$ is the new parent logit, and $c_1, c_2, \\ldots$ are the

476        the following formula $$c' = \\sigma^{-1} ( \\sigma (p) \\sigma (c) )$$
477

480    def _mask(idxs: Iterable[int]) -> Tensor:
481        """
482        Produces a `(n_true_targets,)` binary tensor m, where `m_i` is 1 iff
483        `i` is in `idx`. The tensor is cast to `float` and moved to the the
484        same device as `y`
485        """
486        m = list(map(lambda i: int(i in idxs), range(n_true_targets)))
487        return torch.tensor(m, dtype=torch.float32, device=y.device)
488

490    # 4) means that target 0 (Allergy_Present) is the parent of targets 1 to 4
491    # (Severe_Allergy, Respiratory_Allergy, Food_Allergy, and Venom_Allergy).

493    hierarchy = [  # topmost parent to bottommost children

495        (2, 5, 13),
496        (3, 14, 24),
497        (4, 25, 26),
498    ]

500        hierarchy = hierarchy[::-1]
501    n_true_targets = y.shape[-1]
502    for p, c1, c2 in hierarchy:
503        mc = _mask(range(c1, c2 + 1))  # mc_j = 1 if j child of p
504        if mode == "min":
505            a, b = y[:, [p]].repeat(1, n_true_targets), mc * y
506            c = torch.minimum(a, b)
507            y = c + (1 - mc) * y  # c is already masked
508        elif mode == "max":
509            mp = _mask([p])  # mp_j = 1 if j = p
510            a, b = y[:, p], (mc * y).max(dim=1).values
511            c = torch.maximum(a, b).unsqueeze(-1)
512            y = mp * c + (1 - mp) * y
513        elif mode == "prod":

515            c = -torch.log(torch.exp(-a) + torch.exp(-b) + torch.exp(-a - b))
516            # c = \\sigma^{-1} (\\sigma (a) \\sigma (b))
517            y = c + (1 - mc) * y  # c is already masked
518        else:
519            raise ValueError(
520                f"Unsupported logit correction mode '{mode}'. Available modes "
521                "are 'max', 'min', and 'prod'"
522            )
523    return y
